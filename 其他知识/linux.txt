pwd	: 打印当前工作目录
ls	：打印当前目录的所有文件 -l -al
cd	:进入某个目录
man	：帮助查看命令
clear	:清屏

su xxx	:更改用户
exit	:退出用户
passwd	:更改密码
who	:返回用户信息
groups	：查看用户目前的组
sudo usermod	:usermod -a -G aaa xxx  将xxx加入aaa组
chmod	:chmod +? xxx 对某个文件修改权限 chmod a+? 可以修改全部的权限

cat           :-n xxx 给文件所有行编号
head	:-n x 显示文件中前x行中的内容
diff	:对比文件差异
wc	:统计文件的字节数，字数，行数
mkdir	:制作目录
cp	:复制文件
rm	:删除文件
mv	:移动文件
touch	:创建一个文件
file	:查看文件类型
find	:find -name xxx 查找文件
vim	：在文件中写入内容 按esc后:wq保存文件并退出
du	：du -h xxx 直接显示文件大小
cc	：命令格式 cc xxx.c -o xxx 将c语言文件转换为可执行文件
echo	：shell脚本向命令行中写入内容
./	：运行可执行文件
tar	:tar -c -f xxx.tar *.c 将.c文件压缩
ifconfig	:查看ip地址
ip address show : 查看ip地址

service --status-all|grep xxx:查看xxx服务是否是开启状态
sudo service  xxx start	:开启xxx服务
mysql -u root -p 		:使用mysql服务
apt-get update		:更新apt-get
apt-get install xxx		:更新xxx安装程序
